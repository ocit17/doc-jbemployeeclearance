def autocreate_ad(self, **kwargs):
        c, domain_server = self.connect_toad() 
        domain      = domain_server+'.com'
        username    = kwargs['username']
        useremail   = kwargs['email_private']
        userpswd    = kwargs['userpswd']
        
        check_acc       = self.check_account(username, kwargs['oucomp'], kwargs['department'])
        if check_acc != None:
            username = check_acc['sAMAccountName']
            check_acc2 = self.check_account(check_acc['sAMAccountName'], kwargs['oucomp'], kwargs['department'])
            if check_acc2 != None:
                username = check_acc2['sAMAccountName']

        cek_ou = self.check_ou(kwargs['department'])
        if cek_ou != None:
            dn_ou = cek_ou['dn_ou']

        cek_mgr = self.check_mgr(kwargs['manager'])
        if cek_mgr != None:
            user_manager = cek_mgr['dn_mgr']
        
        userdn      = 'CN='+str(username.replace("."," ").title())+','+dn_ou
        # create user
        create = c.add(userdn, attributes={
            'objectClass': ['organizationalPerson', 'person', 'top', 'user'],
            'sAMAccountName': username,
            'userPrincipalName': "{}@{}".format(username, domain),
            'givenName': kwargs['givenName'],
            'sn':kwargs['sn'],
            'displayName': kwargs['displayName'],
            'mail': "{}@{}".format(username, domain),
            'title':kwargs['level'],
            'department':kwargs['department'],
            'physicalDeliveryOfficeName':kwargs['office'],
            'telephoneNumber':kwargs['phone'],
            'company':kwargs['company'],
        })

        # set password - must be done before enabling user
        # you must connect with SSL to set the password
        c.extend.microsoft.modify_password(userdn, userpswd)

        # cenable user changes password at next logon
        c.modify(userdn, {'pwdLastSet': [('MODIFY_REPLACE', 0)]})

        emp_id = c.modify(userdn, {'employeeID': [('MODIFY_REPLACE', kwargs['empid'])]})
        add_manager = c.modify(userdn, {'manager': [('MODIFY_REPLACE', user_manager)]})
        proxyAddresses = c.modify(userdn, {'proxyAddresses': [('MODIFY_REPLACE', ["smtp:{}@{}".format(username, domain_server+'.onmicrosoft'+'.com'), "SMTP:{}@{}".format(username, domain),])]})

        # enable user (after password set)
        result = c.modify(userdn, {'userAccountControl': [('MODIFY_REPLACE', 512)]})

        return {'create':create, 'modified':add_manager, 'name':kwargs['displayName'], 'mail':"{}@{}".format(username, domain)}
