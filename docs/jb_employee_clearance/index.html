Function auto disable AD

def auto_disable_account_ad(self):
    autodisable_active_directory = self.env['ir.config_parameter'].sudo().search([('key', '=', 'jb_employee_clearance.autodisable_active_directory')]).value
    if bool(autodisable_active_directory) == True:
        date_now =  date.today().isoformat()
        get = self.search([('approved_mgr', '=', True), ('approved_fac', '=', True), ('approved_it', '=', True), ('approved_hr', '=', True)])
        if len(get) > 0 :
            for i in get:
                days_after = (i.resign_confirm_date+timedelta(days=int(i.account_login))).isoformat()
                if days_after == date_now:
                    server = self.env['jb.employee.ad.server'].sudo().active_server()['domain']
                    sAMAccountName = i.employee_id.work_email.replace("@"+server,"")
                    oucomp = i.employee_id.company_id.ou_name
                    dn_name = i.employee_id.name
                    check_ou = self.env['jb.employee.new'].sudo().check_ou(i.department_id.name)
                    if check_ou != None:
                        dn_ou = check_ou['dn_ou']
                    
                    check_acc       = self.env['jb.employee.new'].sudo().check_account(sAMAccountName, oucomp)
                    if check_acc != None and check_acc['data'] > 0:
                        c, domain_server = self.env['jb.employee.new'].sudo().connect_toad() 
                        search_filter = "(sAMAccountName="+sAMAccountName+")" 
                        attrs = ['*']
                        c.search(dn_ou, search_filter, attributes=attrs)
                        data = []
                        for idx, val in enumerate(c.entries):
                            data.append(val.distinguishedName.value)
                        # disable user AD
                        disable_account = c.modify(data[0], {'userAccountControl': [('MODIFY_REPLACE', 2)]})  
                        if disable_account == True:
                            self.env['hr.employee'].sudo().browse(i.employee_id.id).write({'active': False})
